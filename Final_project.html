import pandas as pd
import datetime
import matplotlib.pyplot as plt
import seaborn as sns
import matplotlib.gridspec as gs

%matplotlib notebook

# DATA CLEANING

# India statistics of COVID-19
india = pd.read_csv('india_covid19.csv')
india= india.groupby(['State/UnionTerritory'])['Confirmed'].max()
India = pd.DataFrame(india).reset_index().rename(columns={'State/UnionTerritory':'State / Union Territory','Confirmed':"Total_Cases"})



# Number of Hospital beds in each State of India
hospital_beds = pd.read_csv('HospitalBedsIndia.csv')
hospital_beds['Total_beds'] = hospital_beds.sum(axis=1)
hospital_beds = hospital_beds[['State/UT','Total_beds']]
hospital_beds.rename(columns={'State/UT':'State / Union Territory'}, inplace=True)

# Number of ICMRTestingLabs in each state of India
ICMRTestingLabs = pd.read_csv('ICMRTestingLabs.CSV')
ICMRTestingLabs = ICMRTestingLabs.groupby(['state'])['lab'].count()
ICMRTestingLabs = pd.DataFrame(ICMRTestingLabs).reset_index().rename(columns={'state':'State / Union Territory','lab':'ICMRTestingLabs'})


# Statewise Testing Details in India
StatewiseTesting = pd.read_csv("StatewiseTestingDetails.csv")
StatewiseTesting = StatewiseTesting.groupby(['State'])['TotalSamples'].sum()
StatewiseTesting = pd.DataFrame(StatewiseTesting).reset_index().rename(columns={'State':'State / Union Territory','TotalSamples':'StatewiseTesting'})


# Statewise Populatiom of India according to "population_india_census2011"
population_india = pd.read_csv('population_india_census2011.csv')
population_india = population_india[['State / Union Territory', 'Population']]
population_india = population_india.sort_values(by='State / Union Territory')


COVID_stats = pd.merge(pd.merge(pd.merge(pd.merge(India, hospital_beds, on ='State / Union Territory'),
                  ICMRTestingLabs, on = 'State / Union Territory'),StatewiseTesting, on = 'State / Union Territory'),
                  population_india, on = 'State / Union Territory')
#COVID_stats = COVID_stats.set_index('State / Union Territory', drop=True)
